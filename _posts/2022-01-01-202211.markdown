```C++
class Solution {
public:
    vector<vector<int>> construct2DArray(vector<int>& original, int m, int n) {

        int number = original.size();
        if(m*n != number)
            return {};
        vector<vector<int>> newArray(m,vector<int>(n,0));
        int p=0;
        for(int i=0;i<m;i++)
        {
            for(int j=0;j<n;j++)
            {
                newArray[i][j]=original[i*n+j];
            }
        }
        return newArray;
    }
};
```
